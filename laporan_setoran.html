async function setorDataHariIni() { // 1. Ambil data kalkulasi terbaru const calc = hitungOtomatis(); // 2. Ambil semua input const plat = document.getElementById('selMobil').value; const kmRaw = getAngka(document.getElementById('km_input').value); const checked = Array.from(document.querySelectorAll('input[name="nota_pilih"]:checked')); const pilihanAgen = document.getElementById('sumber_agen').value; // 3. Validasi if(!plat) return alert("Pengecekan Gagal: Unit Armada belum dipilih!"); if(checked.length === 0) return alert("Pengecekan Gagal: Anda belum memilih Nota harian!"); if(!pilihanAgen) return alert("Pengecekan Gagal: Silakan pilih Agen sumber dana!"); if(kmRaw === 0) return alert("Pengecekan Gagal: Angka Odometer KM kosong!"); // 4. Kalkulasi Biaya const uJalan = getAngka(document.getElementById('exp_jalan').value); const uBBM = getAngka(document.getElementById('exp_bbm').value); const uBongkar = getAngka(document.getElementById('exp_bongkar').value); const ketLain = document.getElementById('exp_lain_ket').value || "-"; const nomLain = getAngka(document.getElementById('exp_lain_nom').value); const totalExp = uJalan + uBBM + uBongkar + nomLain; const sisaBersih = calc.rawTotal - totalExp; const stringNoNota = calc.detail.map(d => d.nota).join(', '); const stringNamaAgen = calc.detail.map(d => d.agen).join(', '); try { // Simpan ke database const { error: errorInsert } = await _sb.from('rekap_laporan_driver').insert([{ no_nota: stringNoNota, nama_agen: stringNamaAgen, nama_driver: calc.drv1, driver_2: calc.drv2, plat_nomor: plat, km_input: kmRaw, uang_jalan: uJalan, uang_bbm: uBBM, uang_bongkar: uBongkar, lain_lain: nomLain, ket_lain_lain: ketLain, sumber_agen: pilihanAgen }]); if (errorInsert) throw errorInsert; // Update KM await _sb.from('monitoring_kendaraan').update({ km_saat_ini: kmRaw }).eq('plat_nomor', plat); // 5. FORMAT PESAN WA (Sesuai keinginan Bapak yang komplit) const tgl = new Date().toLocaleDateString('id-ID', { weekday:'long', day:'numeric', month:'long', year:'numeric' }); let pesan = "*LAPORAN SETORAN DRIVER*\n"; pesan += "----------------------------\n"; pesan += "📅 *Tgl:* " + tgl + "\n"; pesan += "👤 *Driver 1:* " + calc.drv1 + "\n"; pesan += "👥 *Driver 2:* " + calc.drv2 + "\n"; pesan += "🚗 *Unit:* " + plat + "\n"; pesan += "📍 *Odo:* " + kmRaw.toLocaleString('id-ID') + " KM\n\n"; pesan += "*RINCIAN BUNDEL CASH:*\n"; calc.detail.forEach(b => { pesan += "- " + b.agen + " (" + b.nota + "): *Rp " + parseInt(b.cash).toLocaleString('id-ID') + "*\n"; }); pesan += "💰 *Total Kotor:* Rp " + calc.rawTotal.toLocaleString('id-ID') + "\n\n"; pesan += "*PENGELUARAN JALAN:*\n"; pesan += "💳 *Beban Kas:* " + pilihanAgen + "\n"; pesan += "- Uang Jalan: Rp " + uJalan.toLocaleString('id-ID') + "\n"; pesan += "- BBM/Solar: Rp " + uBBM.toLocaleString('id-ID') + "\n"; pesan += "- Bongkar: Rp " + uBongkar.toLocaleString('id-ID') + "\n"; pesan += "- Lain-lain (" + ketLain + "): Rp " + nomLain.toLocaleString('id-ID') + "\n"; pesan += "----------------------------\n"; pesan += "📊 *Total Biaya:* Rp " + totalExp.toLocaleString('id-ID') + "\n"; pesan += "💵 *SISA SETOR:* Rp " + sisaBersih.toLocaleString('id-ID') + "\n\n"; pesan += "_Data sukses mengunci ke Supabase SBR_"; window.open("https://wa.me/?text=" + encodeURIComponent(pesan), '_blank'); setTimeout(() => { location.reload(); }, 1500); } catch (e) { alert("⚠️ Error: " + e.message); } }
